-- tables
-- Table: Actor
CREATE TABLE Actor (
    ID int  NOT NULL,
    First_Name varchar(50)  NOT NULL,
    Last_name varchar(50)  NOT NULL,
    CONSTRAINT Actor_pk PRIMARY KEY  (ID)
);

-- Table: Award
CREATE TABLE Award (
    ID int  NOT NULL,
    Movie_ID int  NOT NULL,
    Name varchar(100)  NOT NULL,
    Year date  NOT NULL,
    Award_Category_ID int  NOT NULL,
    Actor_ID int  NULL,
    Director_ID int  NULL,
    CONSTRAINT Award_pk PRIMARY KEY  (ID)
);

-- Table: Award_Category
CREATE TABLE Award_Category (
    ID int  NOT NULL,
    Name varchar(100)  NOT NULL,
    CONSTRAINT Award_Category_pk PRIMARY KEY  (ID)
);

-- Table: Director
CREATE TABLE Director (
    ID int  NOT NULL,
    First_Name varchar(50)  NOT NULL,
    Last_Name varchar(50)  NOT NULL,
    CONSTRAINT Director_pk PRIMARY KEY  (ID)
);

-- Table: Genre
CREATE TABLE Genre (
    ID int  NOT NULL,
    Name varchar(50)  NOT NULL,
    CONSTRAINT Genre_pk PRIMARY KEY  (ID)
);

-- Table: Language
CREATE TABLE Language (
    ID int  NOT NULL,
    Name varchar(50)  NOT NULL,
    CONSTRAINT Language_pk PRIMARY KEY  (ID)
);

-- Table: Movie
CREATE TABLE Movie (
    ID int  NOT NULL,
    Title varchar(100)  NOT NULL,
    Rating int  NOT NULL,
    Director_ID int  NOT NULL,
    Release_date date  NOT NULL,
    Plot varchar(500)  NOT NULL,
    CONSTRAINT Movie_pk PRIMARY KEY  (ID)
);

-- Table: Movie_Genre
CREATE TABLE Movie_Genre (
    Movie_ID int  NOT NULL,
    Genre_ID int  NOT NULL,
    CONSTRAINT Movie_Genre_pk PRIMARY KEY  (Movie_ID,Genre_ID)
);

-- Table: Movie_Language
CREATE TABLE Movie_Language (
    Language_ID int  NOT NULL,
    Movie_ID int  NOT NULL,
    CONSTRAINT Movie_Language_pk PRIMARY KEY  (Language_ID,Movie_ID)
);

-- Table: Rewiev
CREATE TABLE Rewiev (
    ID int  NOT NULL,
    Movie_ID int  NOT NULL,
    Rating int  NOT NULL,
    User_Rewiev varchar(1000)  NULL,
    Date date  NOT NULL,
    User__ID int  NOT NULL,
    CONSTRAINT Rewiev_pk PRIMARY KEY  (ID)
);

-- Table: Role
CREATE TABLE Role (
    Actor_ID int  NOT NULL,
    Movie_ID int  NOT NULL,
    Name varchar(100)  NOT NULL,
    CONSTRAINT Role_pk PRIMARY KEY  (Actor_ID,Movie_ID)
);

-- Table: Trailer
CREATE TABLE Trailer (
    ID int  NOT NULL,
    Movie_ID int  NOT NULL,
    URL varchar(200)  NOT NULL,
    CONSTRAINT Trailer_pk PRIMARY KEY  (ID)
);

-- Table: User_
CREATE TABLE User_ (
    ID int  NOT NULL,
    Username varchar(50)  NOT NULL,
    Email varchar(50)  NOT NULL,
    Password_ varchar(50)  NOT NULL,
    CONSTRAINT User__pk PRIMARY KEY  (ID)
);

-- Table: Watchlist
CREATE TABLE Watchlist (
    ID int  NOT NULL,
    Movie_ID int  NOT NULL,
    User__ID int  NOT NULL,
    CONSTRAINT Watchlist_pk PRIMARY KEY  (ID)
);
